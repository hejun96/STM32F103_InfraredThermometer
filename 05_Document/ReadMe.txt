git status

7.本地提交

git add file

git add .

git conmit -m "This is xxxx."

8.提交到远程

git push

git push -f

9.同步远程的代码

git pull



10.制作补丁

git log

git format-patch -1 6c7769ae8446cae362f4ff0b8d39481835265751

git am xxx.patch

git show-branch
